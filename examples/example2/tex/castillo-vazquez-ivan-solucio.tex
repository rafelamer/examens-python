\begin{enunciat}
Donat el sistema d'equacions
\[
  \left.\aligned x + 2 y + m z &= 3 \\ 3 x + y - 2 z &= m - 12 \\ x + 3 y + 2 z &= 7 \endaligned\;\right\}\,,
\]
\begin{apartats}
\item trobeu el valor o valors de $m$ per al qual el sistema és compatible indeterminat;
\item resoleu-lo per a aquests valors de $m$.
\end{apartats}
\end{enunciat}

\begin{quadricula}
\begin{tabular}{|L{3cm}{1.5cm}|L{7cm}{1.5cm}|}
\hline
$m=$ & \\
\hline
\end{tabular}
\end{quadricula}

\begin{solucio}
\begin{center}
\begin{tabular}{|L{3cm}{1.5cm}|L{7cm}{1.5cm}|}
\hline
$m=1$ & $\left.\aligned x &= z -5 \\ y &= - z + 4 \endaligned\;\right\}$\\
\hline
\end{tabular}
\end{center}
\end{solucio}
