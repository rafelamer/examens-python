\begin{enunciat}
Donat el sistema d'equacions
\[
  \left.\aligned m x + 3 y + z &= -11 \\ 3 x + 3 y + z &= m - 14 \\ - x + y - z &= 1 \endaligned\;\right\}\,,
\]
\begin{apartats}
\item trobeu el valor o valors de $m$ per al qual el sistema és compatible indeterminat;
\item resoleu-lo per a aquests valors de $m$.
\end{apartats}
\end{enunciat}

\begin{quadricula}
\begin{tabular}{|L{3cm}{1.5cm}|L{7cm}{1.5cm}|}
\hline
$m=$ & \\
\hline
\end{tabular}
\end{quadricula}

\begin{solucio}
\begin{center}
\begin{tabular}{|L{3cm}{1.5cm}|L{7cm}{1.5cm}|}
\hline
$m=3$ & $\left.\aligned x &= \frac{- 2 z -7}{3} \\ y &= \frac{z -4}{3} \endaligned\;\right\}$\\
\hline
\end{tabular}
\end{center}
\end{solucio}
